\begin{frame}[fragile]

\frametitle{DEMO: Log}

\begin{itemize}

\item Put something in a file and print it:

\begin{verbatim}
$ echo hello > foo
$ cat foo
hello
\end{verbatim}

\item Use \texttt{strace} to trace system calls:

\begin{verbatim}
$ strace cat foo 
execve(...
...
open("foo", O_RDONLY)
...
\end{verbatim}

\item You can limit \texttt{strace} to trace a handful system calls:

\begin{verbatim}
$ strace -e trace=open,read,write,close cat foo
\end{verbatim}

\end{itemize}

\end{frame}
